用二进制数保存状态（根据题意来确定0/1分别代表的状态，便于位运算），用十进制数来表示二进制。
![](https://cdn.acwing.com/media/article/image/2021/01/19/64341_56cfb8ee5a-%E8%92%99%E5%BE%B7%E9%87%8C%E5%AE%89%E7%9A%84%E6%A2%A6%E6%83%B3-1.png)
```java
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (true) {
            // n:行 m:列 每一列状态总数 2^n
            int n = sc.nextInt(), m = sc.nextInt();
            //输入 0 0结束
            if (n == 0 && m == 0)
              break;
            //最多11行，1 << 12表示每一列最多 2^12 种状态
            long[][] dp = new long[12][1 << 12];
            //每一种状态是否符合去掉1之后连续偶数个0，满足竖着放
            boolean[] st = new boolean[1 << 12];
            //保存有哪些状态 k 可以转移到 j
            ArrayList<ArrayList<Integer>> stage = new ArrayList<>();
            for (int i = 0; i < (1 << n); i++) {
                int cnt = 0;
                boolean flag = true;
                for (int j = 0; j < n; j++) {
                    // 状态i从右开始第j位是1，数前面的0是否满足偶数，不然无法竖着放
                    if (((i >> j) & 1) == 1) {
                        if ((cnt & 1) == 1) {
                            flag = false;
                            break;
                        }
                        cnt = 0;
                    } else
                        cnt++;
                }
                //最后一段是否有奇数个0
                if ((cnt & 1) == 1)
                    flag = false;
                st[i] = flag;
            }
            // 是否冲突
            // 第i-1列伸出到i列的状态j
            for (int j = 0; j < (1 << n); j++) {
                stage.add(new ArrayList<Integer>());
                // 第i-2列伸出到i-1列的状态k
                // 对于k所在的i-1列
                // j和k的1不能在同一行,i-1列的1实际上有j|k中包含的1个数
                for (int k = 0; k < (1 << n); k++) {
                    if ((j & k) == 0 && st[j | k]) {
                        stage.get(j).add(k);
                    }
                }
            }
            // 竖着放
            dp[0][0] = 1;

            for (int i = 1; i <= m; i++) {
                for (int j = 0; j < (1 << n); j++) {
                    for (int k : stage.get(j))
                        dp[i][j] += dp[i - 1][k];
                }
            }
            //0~m-1列是可以摆方块的位置
            System.out.println(dp[m][0]);
        }
    }
}
```
