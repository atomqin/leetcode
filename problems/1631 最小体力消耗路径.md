「二分答案」：我们可以对最短路径的长度进行二分。当我们二分枚举到的长度为 x 时，我们只保留所有长度 ≤x 的边。随后从左上角开始进行搜索（深度优先搜索、广度优先搜索）均可，只需要判断是否能够到达右下角即可。

如果能够到达右下角，我们就可以减小 x 的值，反之增大 x 的值。

「并查集」：我们可以将所有边按照长度进行排序并依次添加进并查集，直到左上角和右下角连通为止。
- Kruskal算法

```java
import java.util.ArrayList;

class Solution {
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        UF uf = new UF(m * n);
        List<int[]> edges = new ArrayList<>();
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                int index = i * n + j;
                if(i < m - 1){
                    edges.add(new int[]{index, index + n, Math.abs(heights[i][j] - heights[i + 1][j])});
                }
                if(j < n - 1){
                    edges.add(new int[]{index, index + 1, Math.abs(heights[i][j] - heights[i][j + 1])});
                }
            }
        }
        Collections.sort(edges, (a, b) -> a[2] - b[2]);
        int ans = 0;
        for(int[] edge : edges){
            int a = edge[0], b = edge[1], w = edge[2];
            uf.union(a, b);
            if(uf.query(0, m * n - 1)){
                ans = w;
                break;
            }
        }
        return ans;
    }
    class UF{
        int[] p;
        int[] size;
        public UF(int n){
            p = new int[n];
            size = new int[n];
            for(int i = 0; i < n; i++){
                p[i] = i;
                size[i] = 1;
            }
        }
        private int find(int x){
            if(x != p[x]) p[x] = find(p[x]);
            return p[x];
        }
        //查询是否连通
        private boolean query(int x, int y){
            return find(x) == find(y);
        }
        private void union(int x, int y){
            int rootx = find(x), rooty = find(y);
            if(rootx == rooty) return;
            if(size[rooty] > size[rootx]){
                int temp = rooty;
                rooty = rootx;
                rootx = temp;
            }
            p[rooty] = rootx;
            size[rootx] += size[rooty];
            //写法二，简单粗暴
            //p[rootx] = p[rooty]
        }
    }
}
```

- 二分 + 并查集

```java
import java.util.ArrayList;

class Solution {
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        int left = 0, right = 1000000;
        int ans = 0;
        while(left <= right){
            int mid = (left + right) >> 1;
            if(findPath(heights, mid)){
                right = mid - 1;
                ans = mid;
            }else
                left = mid + 1;
        }
        return ans;
    }
    private boolean findPath(int[][] heights, int diff){
        int row = heights.length, col = heights[0].length;
        UF uf = new UF(row * col);
        for(int i = 0; i < row; i++){
            for(int j = 0; j < col; j++){
                if(i < row - 1 && Math.abs(heights[i][j] - heights[i + 1][j]) <= diff){
                    uf.union(i * col + j, (i + 1) * col + j);
                }
                if(j < col - 1 && Math.abs(heights[i][j] - heights[i][j + 1]) <= diff){
                    uf.union(i * col + j, i * col + j + 1);
                }
            }
        }
        return uf.query(0, row * col - 1);
    }
    class UF{
        int[] p;
        int[] size;
        public UF(int n){
            p = new int[n];
            size = new int[n];
            for(int i = 0; i < n; i++){
                p[i] = i;
                size[i] = 1;
            }
        }
        private int find(int x){
            if(x != p[x]) p[x] = find(p[x]);
            return p[x];
        }
        private boolean query(int x, int y){
            return find(x) == find(y);
        }
        private void union(int x, int y){
            int rootx = find(x), rooty = find(y);
            if(rootx == rooty) return;
            if(size[rooty] > size[rootx]){
                int temp = rooty;
                rooty = rootx;
                rootx = temp;
            }
            p[rooty] = rootx;
            size[rootx] += size[rooty];
        }
    }
}
```

- 二分+BFS

```java
import java.util.LinkedList;

class Solution {
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        int left = 0, right = 1000000;
        int ans = 0;
        while(left <= right){
            int mid = (left + right) >> 1;
            if(findPath(heights, mid)){
                right = mid - 1;
                ans = mid;
            }else
                left = mid + 1;
        }
        return ans;
    }
    int[][] dir = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    private boolean findPath(int[][] heights, int diff){
        int row = heights.length, col = heights[0].length;
        //别忘了
        boolean[] vis = new boolean[row * col];
        Queue<Integer> q = new LinkedList<>();
        q.offer(0);
        vis[0] = true;
        while(!q.isEmpty()){
            int id = q.poll();
            int x = id / col;
            int y = id % col;
            for(int[] d : dir){
                int nx = x + d[0], ny = y + d[1];
                if(nx >= 0 && nx < row && ny >= 0 && ny < col && !vis[nx * col + ny] && Math.abs(heights[x][y] - heights[nx][ny]) <= diff){
                    if(nx * col + ny == row * col - 1) return true;
                    q.offer(nx * col + ny);
                    vis[nx * col + ny] = true;
                }
            }
        }
        return false;
    }
}
```

- 二分+DFS

```java
import java.util.LinkedList;

class Solution {
    public int minimumEffortPath(int[][] heights) {
        row = heights.length;
        col = heights[0].length;
        int left = 0, right = 1000000;
        int ans = 0;
        while(left < right){
            int mid = (left + right) >> 1;
            //每次要新建一个vis
            boolean[][] vis = new boolean[row][col];
            if(findPath(heights, vis, 0, 0, mid)){
                right = mid;
                // ans = mid;
            }else
                left = mid + 1;
        }
        return left;
    }
    int[][] dir = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
    int row, col;
    private boolean findPath(int[][] heights, boolean[][] vis,int x, int y, int diff){
        if(x == row - 1 && y == col - 1) return true;
        vis[x][y] = true;
        for(int[] d : dir){
            int nx = x + d[0], ny = y + d[1];
            if(nx >= 0 && nx < row && ny >= 0 && ny < col && !vis[nx][ny] && Math.abs(heights[x][y] - heights[nx][ny]) <= diff){
                if(findPath(heights, vis, nx, ny, diff)) return true;
            }
        }
        return false;
    }
}
```

- BFS+动态规划

```java
import java.util.PriorityQueue;

class Solution {
    public int minimumEffortPath(int[][] heights) {
        int m = heights.length, n = heights[0].length;
        int[][] f = new int[m][n];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                f[i][j] = Integer.MAX_VALUE;
            }
        }
        f[0][0] = 0;
        int[][] dir = {{1, 0}, {-1, 0}, {0, -1}, {0, 1}};
        //所有边权值按从小到大排序
        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> a.weight - b.weight);
        pq.offer(new Node(0, 0, 0));
        while(!pq.isEmpty()){
            Node node = pq.poll();
            int x = node.x;
            int y = node.y;
            if(x == m - 1 && y == n - 1) return f[m - 1][n - 1];
            for(int[] d : dir){
                int a = x + d[0], b =  y + d[1];
                if(a < 0 || a > m - 1 || b < 0 || b > n - 1) continue;
                int max = Math.max(f[x][y], Math.abs(heights[a][b] - heights[x][y]));
                if(max >= f[a][b]) continue;
                f[a][b] = max;
                pq.offer(new Node(Math.abs(heights[a][b] - heights[x][y]), a, b));
            }
        }
        return f[m - 1][n - 1];
    }
    class Node{
        int weight, x, y;
        public Node(int weight, int x, int y){
            this.weight = weight;
            this.x = x;
            this.y = y;
        }
    }
}
```
